== Default properties

.{properties-file}
[%collapsible]
====
[source,properties,subs=attributes+]
----
tkit.security-dynamic-test.openapi.enabled=false
onecx.proxy.url-rewrite-rules.".*/test-ui/.*".pattern=test-ui/old#id=(?<id>.+)
onecx.proxy.url-rewrite-rules.".*/test-ui/.*".replace-pattern=/ui/mobile/workorder/details/($id)
onecx.proxy.url-rewrite-rules.".*/test-ui-2/.*".pattern=test-ui-2/old#id=(?<id>.+)#test=(?<test>.+)
onecx.proxy.url-rewrite-rules.".*/test-ui-2/.*".replace-pattern=/ui/mobile/workorder2/details/($id)/($test)
onecx.proxy.url-rewrite-rules.".*/test-ui-3/.*".pattern=test-ui-3/old\\\\?id=(?<id>.+)
onecx.proxy.url-rewrite-rules.".*/test-ui-3/.*".replace-pattern=/ui/mobile/workorder3/details/($id)
----
====
